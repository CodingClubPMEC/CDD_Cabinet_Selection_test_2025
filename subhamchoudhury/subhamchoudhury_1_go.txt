package main

func numberOfChild(n int, k int) int {
	queue := make([]int, n)

	i := 0
	for i := range n {
		queue = append(queue, i)
	}
	for k != 0 {
		for i := range queue {
			if i == n-1 {
				i = 0
			}
			print(i)
		}
		k--
	}
	return i
}

func main() {
	n := 3
	k := 5
	child := numberOfChild(n, k)
	println(child)
}
